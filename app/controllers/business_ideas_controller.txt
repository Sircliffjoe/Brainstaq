class BusinessIdeasController < ApplicationController
  before_action :authenticate_user!

  def new
    @business_idea = BusinessIdea.new
    @categories = Category.all
    @skills = Skill.all
  end

  def create
    @business_idea = current_user.business_ideas.build(business_idea_params)
    # @business_idea = BusinessIdea.new(business_idea_params)
    if @business_idea.save
      @business_idea.generate_ideas
      redirect_to business_ideas_path, notice: 'Business idea was successfully created.'
    else
      render :new
    end
  end

  def index
    @generated_ideas = GeneratedIdea.all
  end

  def show
    @generated_idea = GeneratedIdea.find(params[:id])
  end

  private

  def business_idea_params
    params.require(:business_idea).permit(:category_id, skill_ids: [])
  end
end
