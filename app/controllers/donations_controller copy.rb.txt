        :amount => @amount,
        :donor_id => @donor_id,
        :idea_id => @idea_id,
        :response_reference => @response_reference
        )
      @donor_id = current_user.id
      @idea_id = Idea.find_by_id params[:idea]
      @perk = Perk.find_by_id params[:perk_id]
      
      if @donation.save!
        @idea = @donation.idea
        @perk = Perk.find_by_id(params[:perk_id])

        #   puts "Perk not found"
        # end

      else
        render :new
      end

      render json: {error: 'Perk not found'}
    end
  end
